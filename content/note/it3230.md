> [!NOTE]- Ghi chú
> ```base
> filters:
>   and:
>     - file.hasTag("note")
>     - file.hasTag("university")
>     - subject == link(this.file.name)
> formulas:
>   Tiêu đề: |
>     link(file,title)
>   "": ""
> properties:
>   note.unit:
>     displayName: Bài
>   note.chapter:
>     displayName: Chương
> views:
>   - type: table
>     name: Table
>     order:
>       - chapter
>       - unit
>       - formula.Tiêu đề
>     sort:
>       - property: chapter
>         direction: ASC
>       - property: unit
>         direction: ASC
>     columnSize:
>       formula.Tiêu đề: 400
>   - type: cards
>     name: View
>     order:
>       - formula.Tiêu đề
>       - chapter
>       - unit
>     sort:
>       - property: chapter
>         direction: ASC
>       - property: unit
>         direction: ASC
> 
> ```
